

SELECT I.SUMMARY, L.NAME
FROM ISSUES I
JOIN ISSUESLISTS L ON I.ISSUESLIST_ID = L.ID;



SELECT U.FIRSTNAME, U. LASTNAME, I.SUMMARY
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;



SELECT U.FIRSTNAME, U. LASTNAME, COUNT(*) AS NUMBER_OF_TASKS
FROM USERS U
JOIN ISSUES I ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY I.USER_ID_ASSIGNEDTO
HAVING COUNT(*)>0;
